#include <bits/stdc++.h>
#define ll long long
using namespace std;

int main() {
    ll n, x, p;     // p represents index of upper bound of x
    cin >> n;
    vector <ll> v;      // each new no in p signifies new tower

    for (ll i = 0; i < n; i++) { 
        cin >> x;    
        // upper bound returns pointer to element just >= x
        p = upper_bound (v.begin (), v.end (), x) - v.begin ();

        if (p < v.size ())      // implies x is greater max element of v
            v[p] = x;   // updating value at top of tower
        else
            v.push_back (x);    // creating new tower and adding top
    }
    cout << v.size ();     
    return 0;
}
