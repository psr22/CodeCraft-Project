#include <bits/stdc++.h>
using namespace std;
#define ll long long

int main() {
    ll t = 1, n, k, q, d;
    cin >> t;
    
    while (t--) {
        cin >> n >> k >> q;
        vector <ll> a (k+1), b (k+1);
        a[0] = 0;   b[0] = 0;       // starting pos and time

        for (ll i = 1; i <= k; i ++)
            cin >> a[i];
        for (ll i = 1; i <= k; i ++)
            cin >> b[i];

        while (q--) {   // binary search for (ele in a) just <= d
            cin >> d;   
            ll l = 0, r = k;    // inital left and right

            while (l <= r) {
                ll m = l + (r - l) / 2;     // to avoid overflow
                if (a[m] > d)       
                    r = m - 1;      // since we want ele just <= d
                else    
                    l = m + 1;      
            }   
            ll t = (d - a[r]) * (b[r + 1] - b[r]) / (a[r + 1] - a[r]);
            // a[r] just <= d after loop ends and a[l] > d

            // time taken after b[r] to reach (dist = d) = t
            // t = dist after a[r] / speed of car from a[r] to a[r + 1]
            cout << t + b[r] << " ";
        }       
        cout << "\n";
    }
    return 0;
}
