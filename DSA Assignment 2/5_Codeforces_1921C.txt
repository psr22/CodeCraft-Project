#include <bits/stdc++.h>
using namespace std;
#define ll long long
 
int main() {
    ll t, n, f, a, b;
    cin >> t;
    
    while (t--) {
        cin >> n >> f >> a >> b;
        ll m[n + 1];
        
        m[0] = 0;   // starting at 0 moment
        for (int i = 1; i <= n; i ++)
            cin >> m[i];

        for (int i = 1; i <= n; i ++) {
            if (b <= (m[i] - m[i - 1]) * a)
                f -= b;         // phone off     
            else
                f -= (m[i] - m[i - 1]) * a;     // phone on

            // if it's cheaper to turn phone off, do that
            // else we keep the phone on

            if (f <= 0) {   
                cout << "NO\n";
                break;
            }   // no more messages can be sent  
        }
        if (f > 0)
            cout << "YES\n";
    }
    return 0;
}
