#include <bits/stdc++.h>
using namespace std;
#define ll long long

int main() {
    int t, n, s;
    cin >> t;
    while (t--) {
        cin >> n;
        vector <int> v1, v2;
        
        for (int i=0; i<n; i++) {
            cin >> s;
            v1.push_back (s);
        }    
        v2 = v1;    // v2 stores sorted order of strengths
        sort (v2.begin(), v2.end());
        
        for (auto it : v1) {
            if (it != v2[n-1])      // v2[n-1] => MAX val  
                cout << (it - v2[n-1]) << " ";
            else        // v2[n-2] => 2nd highest val
                cout << (it - v2[n-2]) << " ";
        }
        cout << "\n";
    }
    return 0;
}
