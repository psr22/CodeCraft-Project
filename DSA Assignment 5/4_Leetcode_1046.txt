class Solution {
public:
    int lastStoneWeight(vector<int>& stones) {
        priority_queue <int> pq;        // max heap defined

        for (int i = 0; i < stones.size(); i ++)
            pq.push (stones[i]);        // add all elements from stones

        while (pq.size() > 1) {
            int y = pq.top();       pq.pop();
            // take top 2 elements and pop them from the max heap

            int x = pq.top();       pq.pop();       
            if (x != y)             
                pq.push (y - x);    // push back in max heap if condition satisfied
        }
        if (pq.size() == 1)
            return pq.top();    // return last element left
        return 0;       // no stones left so return 0
    }
};
