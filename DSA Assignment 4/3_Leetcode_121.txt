class Solution {
public:
    int maxProfit(vector<int>& prices) {
        int curr = prices[0], ans = 0;
     
        for (int i = 1; i < prices.size(); i ++) {
            if (curr > prices[i]) 
                curr = prices[i];       // greedy approach
            ans = max (ans, prices[i] - curr);
        }
        return ans;     // maximise the diff, by choosing (max - min) in order
    }
};
